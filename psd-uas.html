Fungsi timestamp#

def df_to_X_y(df, window_size=2):
  X, y = [], []
#   df_np = df.to_numpy()
  for i in range(len(df) - window_size):
    seq_x, seq_y = df[i:i+window_size], df[i+window_size]
    X.append(seq_x)
    y.append(seq_y)
  return np.array(X), np.array(y)
